# Select

Use the following select input element to show selectable list of items.

## Usage

<Demo
  component={(props) => (
    <Select class="w-80" {...props}>
      <option value="0">Option One</option>
      <option value="1">Option Two</option>
      <option value="2">Option Three</option>
    </Select>
  )}
  configurator={[
    {
      name: "label",
      type: "text",
      value: "My Field",
      defaultValue: "",
    },
    {
      name: "corner",
      type: "text",
      value: "",
      defaultValue: "",
    },
    {
      name: "helper",
      type: "text",
      value: "This is a helper message",
      defaultValue: "",
    },
    {
      name: "error",
      type: "text",
      value: "",
      defaultValue: "",
    },
    {
      name: "required",
      type: "boolean",
      value: true,
      defaultValue: false,
    },
    {
      name: "disabled",
      type: "boolean",
      value: false,
      defaultValue: false,
    },
  ]}
  template={(props) => `
    import { Select } from 'nullwind';

    export default function Demo() {
      return (
        <Select ${props} bind={this.field}>
          <option value="0">Option One</option>
          <option value="1">Option Two</option>
          <option value="2">Option Three</option>
        </Select>
      );
    }

`}/>
