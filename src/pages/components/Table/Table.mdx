# Table

Tables serve to distribute information in rows and columns, facilitating the visualization and understanding of data and its correlations.

## Usage

<Demo
  component={Preview}
  template={() => `
    import { Table, Avatar, Badge } from 'nullwind';

    export default function Demo() {
      const users = [
        {
          id: 1,
          name: "John Doe",
          title: "Software Engineer",
          email: "foo@bar.com",
          status: "Active",
        },
        {
          id: 2,
          name: "Jane Smith",
          title: "Product Manager",
          email: "foo@bar.com",
          status: "Active",
        },
        {
          id: 3,
          name: "Michael Johnson",
          title: "Data Analyst",
          email: "foo@bar.com",
          status: "Active",
        },
      ];

      const rows = users.map((user) => (
        <tr>
          <td>
            <Avatar src="/avatar.avif" name={user.name} description="User" />
          </td>
          <td>{user.title}</td>
          <td>{user.email}</td>
          <td>
            <Badge color="success">{user.status}</Badge>
          </td>
          <td class="space-x-2">
            <a href="#" class="text-primary-600 underline">
              Edit
            </a>
          </td>
        </tr>
      ));

      return (
        <Table>
          <thead>
            <tr>
              <th>Name</th>
              <th>Title</th>
              <th>Email</th>
              <th>Status</th>
              <th>Actions</th>
            </tr>
          </thead>
          <tbody>{rows}</tbody>
        </Table>
      );
    }

`}/>
